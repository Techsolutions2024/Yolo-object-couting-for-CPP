cmake_minimum_required(VERSION 3.16)

project(YOLOv8SimpleGridDemo VERSION 1.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(CMAKE_AUTOUIC ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)

# Find Qt6 components
find_package(Qt6 REQUIRED COMPONENTS Core Gui Widgets)

# Create executable from single file
add_executable(YOLOv8SimpleGridDemo demo_simple_grid.cpp)

# Link Qt6 libraries
target_link_libraries(YOLOv8SimpleGridDemo PRIVATE
    Qt6::Core
    Qt6::Gui
    Qt6::Widgets
)

# Set output directory
set_target_properties(YOLOv8SimpleGridDemo PROPERTIES
    WIN32_EXECUTABLE TRUE
    MACOSX_BUNDLE TRUE
)
